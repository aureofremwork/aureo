Logging y Caché

Aprende a implementar logging efectivo y estrategias de caché para optimizar tu aplicación.

Sistema de Logging

Configuración de Logging
// config/logging.php
return [
    'default' => 'stack',
    'channels' => [
        'stack' => [
            'driver' => 'stack',
            'channels' => ['daily', 'slack'],
        ],
        'daily' => [
            'driver' => 'daily',
            'path' => storage_path('logs/aureo.log'),
            'days' => 14,
        ],
        'slack' => [
            'driver' => 'slack',
            'url' => env('LOG_SLACK_WEBHOOK_URL'),
            'username' => 'Aureo Logger',
            'emoji' => ':boom:',
            'level' => 'critical',
        ],
    ],
];
Niveles de Log
// Diferentes niveles de logging
Log::emergency($message);
Log::alert($message);
Log::critical($message);
Log::error($message);
Log::warning($message);
Log::notice($message);
Log::info($message);
Log::debug($message);

Uso del Logger

Logging Básico

public function procesar()
{
    try {
        // Código que puede fallar
        Log::info('Iniciando procesamiento');
        
        $resultado = $this->servicio->procesar();
        
        Log::info('Procesamiento completado', [
            'resultado' => $resultado
        ]);
        
        return $resultado;
    } catch (Exception $e) {
        Log::error('Error en procesamiento', [
            'error' => $e->getMessage(),
            'trace' => $e->getTraceAsString()
        ]);
        
        throw $e;
    }
}

Logging Contextual

Log::withContext([
    'user_id' => auth()->id(),
    'ip' => request()->ip()
]);

Log::channel('slack')->error('Error crítico detectado', [
    'modulo' => 'Pagos',
    'transaccion_id' => $id
]);

Sistema de Caché

Configuración de Caché
// config/cache.php
return [
    'default' => 'redis',
    'stores' => [
        'redis' => [
            'driver' => 'redis',
            'connection' => 'cache',
        ],
        'file' => [
            'driver' => 'file',
            'path' => storage_path('framework/cache'),
        ],
    ],
    'prefix' => 'aureo_cache'
];
Operaciones Básicas
// Almacenar en caché
Cache::put('key', $value, $minutes);

// Obtener de caché
$value = Cache::get('key');

// Obtener o almacenar
$value = Cache::remember('key', $minutes, function () {
    return DB::table('datos')->get();
});

// Eliminar de caché
Cache::forget('key');

Estrategias de Caché

Caché de Consultas

class ProductoRepository
{
    public function getDestacados()
    {
        return Cache::tags(['productos'])->remember('productos.destacados', 60, function () {
            return Producto::where('destacado', true)
                          ->with('categoria')
                          ->get();
        });
    }

    public function actualizarProducto($id, $datos)
    {
        $producto = Producto::find($id);
        $producto->update($datos);
        
        // Limpiar caché relacionada
        Cache::tags(['productos'])->flush();
        
        return $producto;
    }
}

Caché de Vistas

@cache('productos.lista', 60)

Monitoreo y Mantenimiento

Comandos de Mantenimiento
# Limpiar toda la caché
php aureo cache:clear

# Limpiar caché de vistas
php aureo view:clear

# Limpiar logs antiguos
php aureo log:clear
Monitoreo de Rendimiento
// Medir tiempo de operaciones
$tiempo = microtime(true);

// Operación costosa
$resultado = $this->operacionCostosa();

$duracion = microtime(true) - $tiempo;
Log::info('Operación completada', [
    'duracion' => $duracion,
    'memoria' => memory_get_usage(true)
]);
